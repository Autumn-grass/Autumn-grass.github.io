Problem 1¶

Suppose $(x_{1}, y_{1}), (x_{2}, y_{2}), \cdots, (x_{n}, y_{n}) \in \mathbb{R} \times \{0,1\}$. Set \begin{equation*} \begin{split} \sigma(t) &= \frac{1}{1 + e^{-t}} \\ p_{i}(a, b) &= \sigma(a + b x_{i})^{y_{i}} (1 - \sigma(a + b x_{i}))^{1 -y_{i}} \\ l_{i}(a, b) &= -\log p_{i}(a, b) \\ &= -y_{i} \log (\sigma(a + b x_{i})) - (1 - y_{i}) \log((1 - \sigma(a + b x_{i}))) \\ l(a, b) &= \sum_{i=1}^{n} l_{i}(a , b) \end{split} \end{equation*}

(a)¶

\textbf{Proof:} \begin{equation*} \begin{split} \sigma(x) & = \frac{1}{1 + e^{-x}} = (1 + e^{-x})^{-1} \\ \sigma '(x) &= (-1) (1 + e^{-x})^{-2} (0 + e^{-x}(-1)) \\ &=\frac{e^{-x}}{(1 + e^{-x})^{2}} \\ \sigma(x) (1 - \sigma(x)) &= \frac{1}{1+e^{-x}} \left( 1 - \frac{1}{1+e^{-x}}\right) \\& = \frac{1}{1+e^{-x}} \left( \frac{1+e^{-x} - 1}{1+e^{-x}}\right) \\& = \frac{e^{-x}}{(1 + e^{-x})^{2}} \end{split} \end{equation*}

Therefore, we proved that $\sigma '(x) = \sigma(x) (1 - \sigma(x))$.

(b)¶

To get the $\bigtriangledown l_{i}(a, b)$, we derive $\frac{\partial l_{i}(a,b)}{\partial a}$ and $\frac{\partial l_{i}(a,b)}{\partial b}$ respectively.

\begin{equation*} \begin{split} \frac{\partial l_{i}(a,b)}{\partial a} &= -y_{i} \frac{\sigma'(a + bx_{i})}{\sigma(a + bx_{i})}\frac{\partial (a + bx_{i})}{\partial a} - (1 - y_{i})\frac{-\sigma'(a + bx_{i})}{1 - \sigma(a + bx_{i})}\frac{\partial (a + bx_{i})}{\partial a} \\& = -y_{i}\frac{\sigma(a + bx_{i})(1 - \sigma(a + bx_{i}))}{\sigma(a + bx_{i})} + (1 - y_{i})\frac{\sigma(a + bx_{i})(1 - \sigma(a + bx_{i}))}{1 - \sigma(a + bx_{i})} \\& = -y_{i}(1 - \sigma(a + bx_{i})) + (1 - y_{i}) \sigma(a + bx_{i}) \\& = \sigma(a + bx_{i}) - y_{i} \end{split} \end{equation*}

\begin{equation*} \begin{split} \frac{\partial l_{i}(a,b)}{\partial b} &= -y_{i} \frac{\sigma'(a + bx_{i})}{\sigma(a + bx_{i})}\frac{\partial (a + bx_{i})}{\partial b} - (1 - y_{i})\frac{-\sigma'(a + bx_{i})}{1 - \sigma(a + bx_{i})}\frac{\partial (a + bx_{i})}{\partial b} \\& = -y_{i}\frac{\sigma(a + bx_{i})(1 - \sigma(a + bx_{i}))}{\sigma(a + bx_{i})}x_{i} + (1 - y_{i})\frac{\sigma(a + bx_{i})(1 - \sigma(a + bx_{i}))}{1 - \sigma(a + bx_{i})}x_{i} \\& = -y_{i}(1 - \sigma(a + bx_{i}))x_{i} + (1 - y_{i}) \sigma(a + bx_{i})x_{i} \\& = [\sigma(a + bx_{i}) - y_{i}]x_{i} \end{split} \end{equation*}

Then, \begin{equation*} \begin{split} \bigtriangledown l_{i}(a, b) &= \begin{pmatrix} \frac{\partial l_{i}(a,b)}{\partial a}\\ \frac{\partial l_{i}(a,b)}{\partial b} \end{pmatrix} \\& = \begin{pmatrix} \sigma(a + bx_{i}) - y_{i}\\ [\sigma(a + bx_{i}) - y_{i}]x_{i} \end{pmatrix} \\& = (\sigma(a + bx_{i}) - y_{i}) \begin{pmatrix} 1\\ x_{i} \end{pmatrix} \end{split} \end{equation*}

And \begin{equation*} \begin{split} \bigtriangledown ^{2} l_{i}(a, b) &= \begin{pmatrix} \frac{\partial ^{2} l_{i}(a,b)}{\partial a^{2}} & \frac{\partial ^{2} l_{i}(a,b)}{\partial a \partial b}\\ \frac{\partial ^{2} l_{i}(a,b)}{\partial b \partial a} & \frac{\partial ^{2} l_{i}(a,b)}{\partial b^{2}} \end{pmatrix} \\& = \begin{pmatrix} \sigma'(a + bx_{i}) & \sigma'(a + bx_{i}) x_{i}\\ \sigma'(a + bx_{i}) x_{i} & \sigma'(a + bx_{i}) x_{i}^{2} \end{pmatrix} \\& = \sigma'(a + bx_{i}) \begin{pmatrix} 1 & x_{i}\\ x_{i} & x_{i}^{2} \end{pmatrix} \end{split} \end{equation*}

Now we are going to prove $\bigtriangledown ^{2} g$ is positive-semidefinite but not positive-definite, making $g(a, b)$ convex but not strictly convex. First, show the following proposition.

\textbf{[Proposition 1]}

Assume that $g : \mathbb{R}^{2} \mapsto \mathbb{R}$ is differentiable, for every $\mathbf{x}$ and $\mathbf{y}$, if the inequality $$g(\mathbf{y}) \geqslant g(\mathbf{x}) +\bigtriangledown ^{2} g(\mathbf{x}) (\mathbf{y} - \mathbf{x})$$ is satisfied, then $g$ is convex.

\textbf{Proof}:

Let $\mathbf{w}$, $\mathbf{z} \in \mathbb{R}^{2}$, and $0 \leqslant \lambda \leqslant 1$. Denote $\mathbf{x} := \lambda \mathbf{w} + (1 - \lambda)\mathbf{z}$. Then the inequality implies \begin{equation} \label{inequality} g(\mathbf{w}) \geqslant g(\mathbf{x}) +\bigtriangledown ^{2} g(\mathbf{x}) (\mathbf{w} - \mathbf{x})\\ g(\mathbf{z}) \geqslant g(\mathbf{x}) +\bigtriangledown ^{2} g(\mathbf{x}) (\mathbf{z} - \mathbf{x}) \end{equation}

Note that \begin{equation*} \mathbf{w} - \mathbf{x} = (1 - \lambda) (\mathbf{w} - \mathbf{z})\\ \mathbf{z} - \mathbf{x} = \lambda (\mathbf{z} - \mathbf{w}) \end{equation*}

We multiply the first line in \ref{inequality} with $\lambda$ and the second line with $1 - \lambda$, and then add the two inequality in \ref{inequality} together, we obtain \begin{equation*} \begin{split} \lambda g(\mathbf{w}) + (1 - \lambda) g(\mathbf{z}) &\geqslant g (\mathbf{x}) + \lambda \bigtriangledown^{\top} g(\mathbf{x})(1 - \lambda)(\mathbf{w} - \mathbf{z}) + \lambda \bigtriangledown^{\top} g(\mathbf{x})(1 - \lambda)(\mathbf{z} - \mathbf{w}) \\ &= g(\lambda \mathbf{w} + (1 - \lambda)\mathbf{z}) \end{split} \end{equation*} Since $\mathbf{w}$ and $\mathbf{z}$ are arbitrary, this proves $g$ is a convex function.

$\square$

Then deduce $\bigtriangledown ^{2} g$ is positive-semidefinite but not positive-definite, making $g(a, b)$ convex but not strictly convex.

\textbf{Proof}:

Now assume the Hessian $\bigtriangledown ^{2} g$ is positive-semidefinite for all $\mathbf{x} \in \mathbb{R}^{2}$. Then $\forall \mathbf{x}$ and $\mathbf{y} \in \mathbb{R}^{2}$, the Taylor's theorem implies that \begin{equation*} g(\mathbf{y}) = g(\mathbf{x}) + \bigtriangledown ^{\top} g(\mathbf{x})(\mathbf{y} - \mathbf{x}) +\frac{1}{2}(\mathbf{y} - \mathbf{x})^{\top}\bigtriangledown ^{2} g(t\mathbf{x} + (1-t)\mathbf{y}) (\mathbf{y} - \mathbf{x}) \end{equation*} for some $0 \leqslant t \leqslant 1$. Since $\bigtriangledown ^{2} g$ is everywhere positive semi-definite, the quadratic term in this equation is always non-negative. Thus we can estimate \begin{equation*} g(\mathbf{y}) \geqslant g(\mathbf{x}) + \bigtriangledown ^{\top} g(\mathbf{x})(\mathbf{y} - \mathbf{x}) \end{equation*} By \textbf{proposition 1}, $g$ is convex but not strictly convex.

$\square$

(c)¶

From (b), we have \begin{equation*} \bigtriangledown ^{2} l_{i}(a, b) = \begin{pmatrix} \sigma'(a + bx_{i}) & \sigma'(a + bx_{i}) x_{i}\\ \sigma'(a + bx_{i}) x_{i} & \sigma'(a + bx_{i}) x_{i}^{2} \end{pmatrix} \end{equation*}

Based on the definition of null space, let $\bigtriangledown ^{2} l_{i}(a, b) (m_{i}, n_{i})^{\top} = (0, 0)^{\top}$, we have

\begin{equation*} \left\{\begin{matrix} \sigma'(a + bx_{i}) m_{i} + \sigma'(a + bx_{i}) x_{i} n_{i} = 0 & (1)\\ \sigma'(a + bx_{i}) x_{i} m_{i} + \sigma'(a + bx_{i}) x_{i}^{2} n_{i} = 0 & (2) \end{matrix}\right. \end{equation*}

From (1), we have $m_{i} = -x_{i}n_{i}$. Let $n_{i} = 1$, then $m_{i} = -x_{i}$. Normalize $n$ and $m$, we have \begin{equation*} \begin{split} n_{i0} &= \frac{1}{\sqrt{x_{i}^{2}+1}} \\ m_{i0}& = -\frac{x_{i}}{\sqrt{x_{i}^{2}+1}} \end{split} \end{equation*}

Therefore, a basis of the null space $N(\bigtriangledown ^{2} l_{i}(a, b))$ is \begin{equation*} \begin{pmatrix} -\frac{x_{i}}{\sqrt{x_{i}^{2}+1}}\\ \frac{1}{\sqrt{x_{i}^{2}+1}} \end{pmatrix} \end{equation*}

Moreover, we can write the null space as \begin{equation*} N(\bigtriangledown ^{2} l_{i}(a, b)) = \left\{ (m_{i}, n_{i})^{\top} : m_{i} = -\frac{x_{i}}{\sqrt{x_{i}^{2}+1}} t_{i}, n_{i} = \frac{1}{\sqrt{x_{i}^{2}+1}}t_{i}, \text{$t_{i}$ is any value and $t_{i} \in \mathbb{R} $ } \right\} \end{equation*}

(d)¶

Note that $N(\bigtriangledown ^{2} l_{i}(a, b)) = \left\{ (m_{i}, n_{i})^{\top} : m_{i} = -\frac{x_{i}}{\sqrt{x_{i}^{2}+1}} t_{i}, n_{i} = \frac{1}{\sqrt{x_{i}^{2}+1}}t_{i}, t_{i} \in \mathbb{R} \right\}$, then,

\begin{equation*} \begin{split} \bigcap_{i=1}^{n} N(\bigtriangledown ^{2} l_{i}(a, b)) &= \left \{ \left \{ \begin{pmatrix} -\frac{x_{1}}{\sqrt{x_{1}^{2}+1}}t_{1}\\ \frac{1}{\sqrt{x_{1}^{2}+1}}t_{1} \end{pmatrix} \right\} \bigcap \left \{ \begin{pmatrix} -\frac{x_{2}}{\sqrt{x_{2}^{2}+1}}t_{2}\\ \frac{1}{\sqrt{x_{2}^{2}+1}}t_{2} \end{pmatrix} \right\} \bigcap \cdots \bigcap \left \{ \begin{pmatrix} -\frac{x_{n}}{\sqrt{x_{n}^{2}+1}}t_{n}\\ \frac{1}{\sqrt{x_{n}^{2}+1}}t_{n} \end{pmatrix} \right\}\right\} \\& = \left \{ \begin{pmatrix} 0\\ 0 \end{pmatrix}\right \} \end{split} \end{equation*}

Since when $x_{i} \neq x_{j}$, then $ \begin{pmatrix} -\frac{x_{i}}{\sqrt{x_{i}^{2}+1}}\\ \frac{1}{\sqrt{x_{i}^{2}+1}} \end{pmatrix} \neq \begin{pmatrix} -\frac{x_{j}}{\sqrt{x_{j}^{2}+1}}\\ \frac{1}{\sqrt{x_{j}^{2}+1}} \end{pmatrix}$. Now we are going to prove that $ \begin{pmatrix} -\frac{x_{i}}{\sqrt{x_{i}^{2}+1}}t_{i}\\ \frac{1}{\sqrt{x_{i}^{2}+1}}t_{i} \end{pmatrix} = \begin{pmatrix} -\frac{x_{j}}{\sqrt{x_{j}^{2}+1}}t_{j}\\ \frac{1}{\sqrt{x_{j}^{2}+1}}t_{j} \end{pmatrix}$ only when they are $0$ vector.

If $-\frac{x_{i}}{\sqrt{x_{i}^{2}+1}}t_{i} = -\frac{x_{j}}{\sqrt{x_{j}^{2}+1}}t_{j} \neq 0$, then we have $t_{i} = \frac{x_{j}}{\sqrt{x_{j}^{2}+1}} \frac{\sqrt{x_{i}^{2}+1}}{x_{i}}t_{j}$. Plug this $t_{i}$ into $\frac{1}{\sqrt{x_{i}^{2}+1}}t_{i}$, we have $\frac{1}{\sqrt{x_{i}^{2}+1}}t_{i} = \frac{1}{\sqrt{x_{i}^{2}+1}}\frac{x_{j}}{\sqrt{x_{j}^{2}+1}} \frac{\sqrt{x_{i}^{2}+1}}{x_{i}}t_{j} = \frac{x_{j}}{x_{i}\sqrt{x_{j}^{2}+1}}t_{j} \neq \frac{1}{\sqrt{x_{j}^{2}+1}}t_{j}$. Therefore, we can not find $t_{i}$ and $t_{j}$ such that $ \begin{pmatrix} -\frac{x_{i}}{\sqrt{x_{i}^{2}+1}}t_{i}\\ \frac{1}{\sqrt{x_{i}^{2}+1}}t_{i} \end{pmatrix} = \begin{pmatrix} -\frac{x_{j}}{\sqrt{x_{j}^{2}+1}}t_{j}\\ \frac{1}{\sqrt{x_{j}^{2}+1}}t_{j} \end{pmatrix} \neq 0$.

(e)¶

From (b) we have \begin{equation*} \bigtriangledown ^{2} l_{i}(a, b) = \begin{pmatrix} \sigma'(a + bx_{i}) & \sigma'(a + bx_{i}) x_{i}\\ \sigma'(a + bx_{i}) x_{i} & \sigma'(a + bx_{i}) x_{i}^{2} \end{pmatrix} \end{equation*}

Then $\forall \mathbf{s} = (s_{1}, s_{2})^{\top} \neq 0$, we have \begin{equation*} \begin{split} \mathbf{s}^{\top} \bigtriangledown ^{2} l_{i}(a, b) \mathbf{s} &= (s_{1}, s_{2}) \begin{pmatrix} \sigma'(a + bx_{i}) & \sigma'(a + bx_{i}) x_{i}\\ \sigma'(a + bx_{i}) x_{i} & \sigma'(a + bx_{i}) x_{i}^{2} \end{pmatrix} \begin{pmatrix} s1\\ s2 \end{pmatrix} \\& = \left(s_{1}\sigma'(a + bx_{i})+ s_{2} \sigma'(a + bx_{i}) x_{i} , s_{1}\sigma'(a + bx_{i}) x_{i} + s_{2}\sigma'(a + bx_{i}) x_{i}^{2} \right) \begin{pmatrix} s1\\ s2 \end{pmatrix} \\& = s_{1}^{2}\sigma'(a + bx_{i})+ s_{2}s_{1} \sigma'(a + bx_{i}) x_{i} + s_{1}s_{2}\sigma'(a + bx_{i}) x_{i} + s_{2}^{2}\sigma'(a + bx_{i}) x_{i}^{2} \\& = s_{1}^{2}\sigma'(a + bx_{i}) + 2 s_{1}s_{2}\sigma'(a + bx_{i}) x_{i} + s_{2}^{2}\sigma'(a + bx_{i}) x_{i}^{2} \\& = \sigma'(a + bx_{i}) (s_{1} + s_{2}x_{i})^{2} \\& > 0 \end{split} \end{equation*} Thus we have $\bigtriangledown ^{2} l_{i}(a, b)$ is positive definite. Since $\bigtriangledown ^{2} l(a, b) = \sum_{i=1}^{n}\bigtriangledown ^{2} l_{i}(a, b)$, therefore, $\bigtriangledown ^{2} l(a, b)$ is also positive definite, since \begin{equation*} \mathbf{s}^{\top} \bigtriangledown ^{2} l(a, b) \mathbf{s} = \sum_{i=1}^{n} \mathbf{s}^{\top} \bigtriangledown ^{2} l_{i}(a, b) \mathbf{s}>0 \end{equation*}

Then by the result in (b), $l(a,b)$ is a convex function.

(f)¶

From (e), for $x_{i} \neq x_{j}$, we have $l(a, b)$ is a convex function. Suppose there are two optimal solutions $\mathbf{\omega} = (a_{1}, b_{1})^{\top}$, $\mathbf{\nu} = (a_{2}, b_{2})^{\top} \in \mathbb{R^{2}}$ s.t.

\begin{equation} \label{l_function} l(\mathbf{\omega}) = l(\mathbf{\nu}) \leqslant l(\mathbf{z}), \ \forall \mathbf{z} \in \mathbb{R}^{2} \end{equation} Let $\mathbf{z} = \frac{\omega + \nu}{2}$, by convexity of function $l(a, b)$, we have \begin{equation} \label{l_inequality} \begin{split} l(\mathbf{z}) = l(\frac{\omega + \nu}{2}) &= l(\frac{1}{2}\omega + \frac{1}{2} \nu) \\& \leqslant \frac{1}{2}l(\omega) + \frac{1}{2}l( \nu) \\& \leqslant \frac{1}{2}l(\omega) + \frac{1}{2}l(\omega) \\& = l(\omega) \end{split} \end{equation}

But this contradicts with \ref{l_function}. Therefore, the minimizer of $l(a, b)$ is unique, which means the maximize likelihood estimates $\hat{a}$, $\hat{b}$ of function $-l(a, b)$ is unique. Note that the second inequality above is because the $\nu$ is arbitrary so we can take $\nu = \omega$.

\textbf{Proof}:

We have \begin{equation*} \begin{split} \\ \sigma '(x) &= (-1) (1 + e^{-x})^{-2} (0 + e^{-x}(-1)) \\ &=\frac{e^{-x}}{(1 + e^{-x})^{2}} \\ & = \sigma(x) (1 - \sigma(x)) \\& > 0 \end{split} \end{equation*} Then $\sigma(x)$ is increasing. Take $x$ approximate to $-\infty$ and $\infty$, we can get the range of \begin{equation*} \sigma(x) = \frac{1}{1 + e^{-x}} \in (0,1) \end{equation*}

Secondly, $\log(x)$ is also increasing in $(0, \infty)$. Specifically, $\log(1) = 0$. Therefore, the range of $\log(x)$ in $(0,1)$ is $(-\infty, 0)$.

Since \begin{equation*} \begin{split} l_{i}(a, b) &= -\log p_{i}(a, b) \\ &= -y_{i} \log (\sigma(a + b x_{i})) - (1 - y_{i}) \log(1 - \sigma(a + b x_{i})) \end{split} \end{equation*}

For $y_{i} = 0$, we have $l_{i}(a, b) = - \log(1 - \sigma(a + b x_{i})) \in (0, \infty)$, since $1 - \sigma(a + b x_{i}) \in (0,1)$.

For $y_{i} = 1$, we have $l_{i}(a, b) = - \log (\sigma(a + b x_{i})) \in (0, \infty)$, since $\sigma(a + b x_{i}) \in (0,1)$.

From the above, $l_{i}(a, b) > 0$ for all $(a, b) \in \mathbb{R}^{2}$.

(b)¶

We have \begin{equation*} \begin{split} l_{i}(t v_{1}, t v_{2}) = -y_{i} \log (\sigma(t v_{1} + t v_{2} x_{i})) - (1 - y_{i}) \log(1 - \sigma(t v_{1} + t v_{2} x_{i})) \end{split} \end{equation*}

When $y_{i} = 0$, $l_{i}(t v_{1}, t v_{2}) = - \log(1 - \sigma(t v_{1} + t v_{2} x_{i}))$, then \begin{equation*} \begin{split} H_{i} &= \{(v_{1}, v_{2})^{\top} \in \mathbb{R}^{2} : \lim_{t \to \infty} l_{i}(t v_{1}, t v_{2}) = \infty \} \\& = \{(v_{1}, v_{2})^{\top} \in \mathbb{R}^{2} : \lim_{t \to \infty} [-\log(1 - \sigma(t v_{1} + t v_{2}x_{i}))] = \infty \} \\& = \{(v_{1}, v_{2})^{\top} \in \mathbb{R}^{2} : \lim_{t \to \infty} (1 - \sigma(t v_{1} + t v_{2}x_{i})) = 0 \} \\& = \{(v_{1}, v_{2})^{\top} \in \mathbb{R}^{2} : \lim_{t \to \infty} (\sigma(t v_{1} + t v_{2}x_{i})) = 1 \} \\& = \{(v_{1}, v_{2})^{\top} \in \mathbb{R}^{2} : \lim_{t \to \infty} t (v_{1} + v_{2}x_{i}) = \infty \} \\& = \{(v_{1}, v_{2})^{\top} \in \mathbb{R}^{2} : v_{1} + v_{2}x_{i} > 0 \} \\& = \{(v_{1}, v_{2})^{\top} \in \mathbb{R}^{2} : (v_{1}, v_{2}) \cdot (1, x_{i})^{\top} > 0 \} \\& = H(w_{i}) \end{split} \end{equation*} where $w_{i} = (1, x_{i})^{\top}$.

When $y_{i} = 1$, $l_{i}(t v_{1}, t v_{2}) = - \log(\sigma(t v_{1} + t v_{2} x_{i}))$, then \begin{equation*} \begin{split} H_{i} &= \{(v_{1}, v_{2})^{\top} \in \mathbb{R}^{2} : \lim_{t \to \infty} l_{i}(t v_{1}, t v_{2}) = \infty \} \\& = \{(v_{1}, v_{2})^{\top} \in \mathbb{R}^{2} : \lim_{t \to \infty} [-\log(\sigma(t v_{1} + t v_{2}x_{i}))] = \infty \} \\& = \{(v_{1}, v_{2})^{\top} \in \mathbb{R}^{2} : \lim_{t \to \infty} \sigma(t v_{1} + t v_{2}x_{i}) = 0 \} \\& = \{(v_{1}, v_{2})^{\top} \in \mathbb{R}^{2} : \lim_{t \to \infty} t (v_{1} + v_{2}x_{i}) = - \infty \} \\& = \{(v_{1}, v_{2})^{\top} \in \mathbb{R}^{2} : v_{1} + v_{2}x_{i} < 0 \} \\& = \{(v_{1}, v_{2})^{\top} \in \mathbb{R}^{2} : - v_{1} - v_{2}x_{i} > 0 \} \\& = \{(v_{1}, v_{2})^{\top} \in \mathbb{R}^{2} : (v_{1}, v_{2}) \cdot (-1, -x_{i})^{\top} > 0 \} \\& = H(w_{i}) \end{split} \end{equation*} where $w_{i} = (-1, -x_{i})^{\top}$.

In conclusion, $w_{i} = (-1)^{y_{i}}\cdot(1, x_{i})^{\top}$.
